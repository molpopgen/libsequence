68  FSTimpl::FSTimpl(const PolyTable *data, unsigned npop, const unsigned *config,
69  const double *weights, bool haveOutgroup, unsigned outgroup) :
70  _Counts( _vvCounts(npop,_vCounts(data->numsites()))),
71  _nsam(unsigned(data->size())),
72  _nsites(data->numsites()),_npop(npop),
73  _piB(0.), _piT(0.), _piS(0.), _piD(0.),
74  _indexes( vector< pair<unsigned,unsigned> >(npop) ),
75  _calcsDone(false),
76  pv( make_polySiteVector(*data) )
77  {
78  if (config == NULL)
79  {
80  throw SeqException("Seqence::FST -- config vector is NULL");
81  }
82  else
83  {
84  _config.assign(config,config+npop);
85  if(accumulate(_config.begin(),_config.end(),
86  unsigned(0))+unsigned(haveOutgroup) != _nsam)
87  {
88  throw SeqException("Seqence::FST -- sum of elements in config does not equal the sample size stored in data");
89  }
90  }
91  if (weights == NULL)
92  //equally weight all populations
93  {
94  _weights.assign(_npop, 1./double(_npop));
95  }
96  else
97  {
98  _weights.assign(weights,weights+npop);
99  double sum = accumulate(_weights.begin(),_weights.end(),0.);
100  if ( std::fabs(sum-1.) > std::numeric_limits<double>::epsilon() )
101  {
102  throw SeqException("Seqence::FST -- weights do not sum to 1");
103  }
104  }
105 
106  _indexes[0] = pair<unsigned,unsigned>(0,_config[0]);
107  for(unsigned i = 1 ; i < _npop ; ++i)
108  {
109  _indexes[i] = pair<unsigned,unsigned>(_indexes[i-1].second,
110  _indexes[i-1].second+_config[i]);
111  }
112  for (unsigned i = 0 ; i < _npop ; ++i) //over pops
113  {
114  for(unsigned j = 0 ; j < _nsites ; ++j)//over sites
115  {
116  _Counts[i][j] = stateCounter('-');
117  for(unsigned k = _indexes[i].first ; k < _indexes[i].second ; ++k)//over sequences
118  {
119  if (haveOutgroup == false ||
120  (haveOutgroup == true && k != outgroup))
121  _Counts[i][j]((*data)[k][j]);
122  }
123  }
124  }
125 
126  if (haveOutgroup == true)
127  {
128  //remove outgroup state from pv
129  for(unsigned i=0;i<pv.size();++i)
130  {
131  pv[i].second.erase(outgroup,1);
132  }
133  }
134  this->doCalcs();
135  }

203  void FSTimpl::doCalcs(void)
204  {
205  //calculate sums of w/in population weights and heterozygosity
206  double w_ii_sq = 0., weighted_Pi_ii = 0.;
207 
208  for (unsigned i = 0 ; i < _npop ; ++i) //over pops
209  {
210  double Pi = 0.;
211  w_ii_sq += _weights[i]*_weights[i];
212  for(unsigned j = 0 ; j < _nsites ; ++j)//over sites
213  {
214  unsigned n = _config[i];
215  double SSH = 0.;
216  n -= _Counts[i][j].n;
217  double denom = (double(n)* (double(n) - 1.0));
218  SSH += (_Counts[i][j].a > 0) ? double(_Counts[i][j].a) *
219  double (_Counts[i][j].a-1) /denom : 0. ;
220  SSH += (_Counts[i][j].g > 0) ? double(_Counts[i][j].g) *
221  double (_Counts[i][j].g-1) /denom : 0. ;
222  SSH += (_Counts[i][j].c > 0) ? double(_Counts[i][j].c) *
223  double (_Counts[i][j].c-1) /denom : 0. ;
224  SSH += (_Counts[i][j].t > 0) ? double(_Counts[i][j].t) *
225  double (_Counts[i][j].t-1) /denom : 0. ;
226  SSH += (_Counts[i][j].zero > 0) ? double(_Counts[i][j].zero) *
227  double (_Counts[i][j].zero-1) /denom : 0. ;
228  SSH += (_Counts[i][j].one > 0) ? double(_Counts[i][j].one) *
229  double (_Counts[i][j].one-1) /denom : 0. ;
230  Pi += (1.0 - SSH);
231  }//sites
232  weighted_Pi_ii += _weights[i]*_weights[i]*Pi;
233  }//pops
234 
235  //now calculate between-population divergence,
236  //using stored state info in _Counts
237  double sum_wi_wj = 0., weighted_Pi_ij = 0.;
238  for(unsigned i = 0 ; i < _npop - 1 ; ++i)//over pops_i
239  {
240  for(unsigned j = i+1 ; j < _npop ; ++j)//over pops_j
241  {
242  double Pi_i_j = 0.;
243  sum_wi_wj += _weights[i]*_weights[j];
244  for(unsigned k = 0 ; k < _nsites ; ++k)//over sites
245  {
246  unsigned ni = _config[i],nj = _config[j];
247  ni -= _Counts[i][k].n;
248  nj -= _Counts[j][k].n;
249  Pi_i_j += (_Counts[i][k].a > 0) ?
250  (double(_Counts[i][k].a)/double(ni))*
251  (double(nj-_Counts[j][k].a)/double(nj)):0.;
252  Pi_i_j += (_Counts[i][k].g > 0) ?
253  (double(_Counts[i][k].g)/double(ni))*
254  (double(nj-_Counts[j][k].g)/double(nj)):0.;
255  Pi_i_j += (_Counts[i][k].c > 0) ?
256  (double(_Counts[i][k].c)/double(ni))*
257  (double(nj-_Counts[j][k].c)/double(nj)):0.;
258  Pi_i_j += (_Counts[i][k].t > 0) ?
259  (double(_Counts[i][k].t)/double(ni))*
260  (double(nj-_Counts[j][k].t)/double(nj)):0.;
261  Pi_i_j += (_Counts[i][k].zero > 0) ?
262  (double(_Counts[i][k].zero)/double(ni))*
263  (double(nj-_Counts[j][k].zero)/double(nj)):0.;
264  Pi_i_j += (_Counts[i][k].one > 0) ?
265  (double(_Counts[i][k].one)/double(ni))*
266  (double(nj-_Counts[j][k].one)/double(nj)):0.;
267  }//sites
268  weighted_Pi_ij += _weights[i]*_weights[j]*Pi_i_j;
269  }//pops_j
270  }//pops_i
271 
272  //calculate measures of diversity/divergence needed to obtain Fst
273  _piT = weighted_Pi_ii + 2.*weighted_Pi_ij;
274  _piS = weighted_Pi_ii / w_ii_sq ;
275  _piB = weighted_Pi_ij / sum_wi_wj;
276  _piD = (_piT - _piS)/(2. * sum_wi_wj);
277 
278  _calcsDone = true;
279  }

281  std::set<double> FST::shared(unsigned pop1, unsigned pop2) const
288  {
289  if (pop1 > impl->_npop-1 || pop2 > impl->_npop-1)
290  throw SeqException("Seqence::FST -- indexes out of range");
291 
292  std::set<double> sharedList;
293 
294  for(unsigned site = 0 ; site < impl->_nsites ; ++site)
295  {
296  if (impl->_Counts[pop1][site].gap == 0 &&
297  impl->_Counts[pop2][site].gap == 0)
298  {
299  FSTimpl::PopStateSets states = impl->getPopStateSets(pop1,pop2,site);
300  if (states.first.size() > 1 && states.second.size() > 1)
301  {
302  vector<char> overlap(states.first.size()+states.second.size());
303  vector<char>::iterator itr = std::set_intersection(states.first.begin(),
304  states.first.end(),
305  states.second.begin(),
306  states.second.end(),
307  overlap.begin());
308  if ( itr - overlap.begin() > 0) //shared states exist
309  {
310  sharedList.insert(impl->pv[site].first);
311  }
312  }
313  }
314  }
315  return sharedList;
316  }

350  std::pair< std::set<double>,std::set<double> >
351  FST::Private(unsigned pop1, unsigned pop2) const
361  {
362  if (pop1 > impl->_npop-1 || pop2 > impl->_npop-1)
363  throw SeqException("Seqence::FST -- indexes out of range");
364 
365  std::set<double> p1,p2;
366  for(unsigned site = 0 ; site < impl->_nsites ; ++site)
367  {
368  if (impl->_Counts[pop1][site].gap == 0 &&
369  impl->_Counts[pop2][site].gap == 0)
370  {
371  FSTimpl::PopStateSets states = impl->getPopStateSets(pop1,pop2,site);
372 
373  //now, use std::set_difference to identify private polys
374  vector<char> priv1(states.first.size()),priv2(states.second.size());
375  //private in pop1
376  vector<char>::iterator itr1 = std::set_difference(states.first.begin(),states.first.end(),
377  states.second.begin(),states.second.end(),
378  priv1.begin());
379  //private in pop2
380  vector<char>::iterator itr2 = std::set_difference(states.second.begin(),states.second.end(),
381  states.first.begin(),states.first.end(),
382  priv2.begin());
383  //if there are unique states && site is polymorphic,
384  //then there is a private poly
385  if ( (itr1-priv1.begin())>0 && states.first.size()>1)
386  {
387  p1.insert(impl->pv[site].first);
388  }
389  if ( (itr2-priv2.begin())>0 && states.second.size()>1)
390  {
391  p2.insert(impl->pv[site].first);
392  }
393  }
394  }
395  return std::make_pair(p1,p2);
396  }
