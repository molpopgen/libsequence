41  /**
42  @ingroup stats
43  Calculates Pearson's product-moment correlation between two containers, \c x and \c y.
44  \f[r_{1,2}=\frac{y_1\times y_2}{\sqrt{\sum_{i=1}^{i=n} y_1^2 \times \sum \sum_{i=1}^{i=n} y_2^2}}\f]
45  From Sokal and Rohlf. Biometry, 3rd ed. 1995, Freeman Press, page 571.
46  @param beg_x pointer to beginning of \c x
47  @param end_x pointer to one past the end of \c x
48  @param beg_y pointer to beginning of \c y
49  @note the square of the return value is the well-know coefficient of determination, r^2
50  */
51  template<typename iter1, typename iter2>
52  double ProductMoment::operator()(iter1 beg_x, iter1 end_x, iter2 beg_y) const
53  {
54  typedef double rtype;
55  if (beg_x >= end_x) return std::numeric_limits<rtype>::min();
56 
57  unsigned nsam = 0;
58  rtype x_bar=rtype();
59  rtype y_bar=rtype();
60  rtype x_squared=rtype();
61  rtype y_squared=rtype();
62  rtype xy=rtype();
63 
64  for ( ; beg_x != end_x ; ++beg_x,++beg_y)
65  {
66  x_bar += *beg_x;
67  x_squared += (*beg_x)*(*beg_x);
68  y_bar += *beg_y;
69  y_squared += (*beg_y)*(*beg_y);
70  xy += (*beg_x)*(*beg_y);
71  ++nsam;
72  }
73  x_bar /= rtype(nsam);
74  y_bar /= rtype(nsam);
75  x_squared /= rtype(nsam);
76  y_squared /= rtype(nsam);
77  xy -= x_bar*rtype(nsam)*y_bar;
78  xy -= y_bar*rtype(nsam)*x_bar;
79  xy += rtype(nsam)*x_bar*y_bar;
80  rtype sum_x_sq = rtype(nsam)*(x_squared-x_bar*x_bar);
81  rtype sum_y_sq = rtype(nsam)*(y_squared-y_bar*y_bar);
82  return(xy/std::pow((sum_x_sq*sum_y_sq),0.5));
83  }

85  template<typename iter1, typename iter2>
86  double SpearmansRank::operator()(iter1 beg_x, iter1 end_x, iter2 beg_y) const
87  {
88  typedef double rtype;
89 
90  if(beg_x>=end_x) return std::numeric_limits<rtype>::min();
91 
92  typedef typename std::iterator_traits<iter1>::value_type t1;
93  typedef typename std::iterator_traits<iter2>::value_type t2;
94  typedef typename std::iterator_traits<iter1>::difference_type dtype1;
95  //typedef typename std::iterator_traits<iter2>::difference_type dtype2;
96  dtype1 d = end_x-beg_x;
97  std::vector<t1> xranks;
98  std::vector<t2> yranks;
99  //get the unique elements in each range
100  std::unique_copy(beg_x,end_x,std::back_inserter(xranks));
101  std::unique_copy(beg_y,beg_y+size_t(d),std::back_inserter(yranks));
102  //sort them
103  std::sort(xranks.begin(),xranks.end());
104  std::sort(yranks.begin(),yranks.end());
105 
106  //Recode the data using a map<value,rank>,
107  //as it's a quick way to do the rank lookups
108  std::map<t1,unsigned> mx;
109  std::map<t2,unsigned> my;
110  unsigned rank=0;
111  for (typename std::vector<t1>::const_iterator i =
112  xranks.begin() ;
113  i != xranks.end() ;
114  ++i)
115  {
116  mx[*i] = rank++;
117  }
118  rank=0;
119  for (typename std::vector<t2>::const_iterator i =
120  yranks.begin() ;
121  i != yranks.end() ;
122  ++i)
123  {
124  my[*i] = rank++;
125  }
126  //calculate Pearson's rho on the ranks
127  unsigned nsam = 0;
128  rtype x_bar=rtype();
129  rtype y_bar=rtype();
130  rtype x_squared=rtype();
131  rtype y_squared=rtype();
132  rtype xy=rtype();
133  typename std::vector<t1>::const_iterator a;
134  typename std::vector<t2>::const_iterator b;
135  for( ; beg_x != end_x ; ++beg_x,++beg_y )
136  {
137  //find where each element is in the ranks vector
138  rtype ra = mx[*beg_x];
139  rtype rb = my[*beg_y];
140  x_bar += ra;
141  x_squared += ra*ra;
142  y_bar += rb;
143  y_squared += rb*rb;
144  xy += ra*rb;
145  ++nsam;
146  }
147  x_bar /= rtype(nsam);
148  y_bar /= rtype(nsam);
149  x_squared /= rtype(nsam);
150  y_squared /= rtype(nsam);
151  xy -= x_bar*rtype(nsam)*y_bar;
152  xy -= y_bar*rtype(nsam)*x_bar;
153  xy += rtype(nsam)*x_bar*y_bar;
154  rtype sum_x_sq = rtype(nsam)*(x_squared-x_bar*x_bar);
155  rtype sum_y_sq = rtype(nsam)*(y_squared-y_bar*y_bar);
156  return(xy/std::pow((sum_x_sq*sum_y_sq),0.5));
157  }

192  template<typename iter1, typename iter2,
193  typename correlation_type,
194  typename comparison_function,
195  typename shuffler>
196  double PermuteCorrelation(iter1 beg_x, iter1 end_x, iter2 beg_y,
197  const correlation_type & corr,
198  const comparison_function & comp,
199  shuffler & s,
200  const unsigned & NPERM)
201  /*!
202  Obtain the p-value of a correlation coefficient by permutation. This function
203  can be used to get 1- or 2- tailed p-values by using different comparison_function
204  objects. For example, using std::greater_equal<double> will returned the 1-tailed
205  probability of observing a correlation >= the observed value.
206  @param beg_x pointer to the beginning of the range of the 1st vector
207  @param end_x pointer to the end of the range of the 1st vector
208  @param beg_y pointer to the beginning of the range of the 2nd vector
209  @param corr a function object to calculate the correlation statistic (i.e. ProductMoment)
210  @param comp a comparison function
211  @param NPERM number of permutations to do
212  @param rand a function returning a random integer (must be compatible with std::random_shuffle)
213  @note This function keeps the order of the 2 containers intact.
214  */
215  {
216  return PermuteCorrelation_details(beg_x,end_x,beg_y,corr,comp,s,NPERM);
217  }
