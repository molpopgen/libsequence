47  /*!
48  Read objects of type T and put them into the vector seqarray.
49  Note that seqarray is not const, so that's where the data go.
50  infilename refers to a file from which to read the data. Any
51  exceptions that can be thrown when reading the data are not
52  caught here.
53  \n
54  Example:\n
55  \code
56  #include <iostream>
57  #include <Sequence/Fasta.hpp>
58  #include <Sequence/Alignment.hpp>
59  //using namespace std;
60  using namespace Sequence;
61  int main
62  {
63  vector<Fasta> data;
64  const char *infile = foo;
65  try {
66  Alignment::GetData(&data,foo);
67  } catch (std::runtime_error &e) {
68  cerr << "whoa--exception caught"<<endl;
69  e.print(cerr);
70  cerr << endl;
71  }
72  }
73  \endcode
74  \param seqarray a vector<T> that you want filled
75  \param infilename name of file from which fo fill seqarray
76  \note if \a infilename is NULL, the function returns, having done nothing
77  */
78  {
79  if(infilename == nullptr)
80  return;
81 
82  std::ifstream infile(infilename);
83  if (infile)
84  {
85  T temp;
86  while (!(infile.eof ()))
87  {
88  infile >> temp >> std::ws;
89  seqarray.push_back(temp);
90  }
91  }
92  }

118  template < typename T >
119  std::istream & ReadNObjects ( std::vector < T > &seqarray,
120  const unsigned & n,
121  std::istream & input_stream)
122  /*!
123  Read a fixed number n of objects of type T and put
124  them into the vector seqarray.
125  Note that seqarray is not const, so that's where the data go.
126  This function is similar to GetData(vector<T> &seqarray,
127  const char *infilename), except that it is passed a reference
128  to an open input stream, such as a file stream, cin, etc.
129  \param seqarray an empty vector<T> that you want filled
130  \param n number of objects of type T to read
131  \param input_stream name of istream from which fo fill seqarray
132  \note The stream is not closed after each read of n records, and
133  seqarray is appended to, and thus grows each time.
134  */
135  {
136  for(unsigned i = 0 ; !input_stream.eof() && i < n ; ++i)
137  {
138  if (!input_stream || input_stream.eof())
139  return input_stream;
140  T temp;
141  input_stream >> temp >> std::ws;
142  seqarray.push_back(temp);
143  }
144  return input_stream;
145  }

193  \return true if each element in the range [beg,end) only contains
194  characters in the set {A,G,C,T,N,-}, false otherwise
195  */
196  {
197  //ensure that Iterator is an iterator type for a container

212  {
213  return false;
214  }

221  unsigned UnGappedLength (const std::vector <T>&data)
222 
223  /*!
224  Returns the number of sites in the alignment for which
225  all objects do not contain the gap character '-'. If the data are not
226  aligned, the value Sequence::SEQMAXUNSIGNED is returned as an error.
227  \param data vector<T> to check
228  */
229  {
230  static_assert( std::is_base_of<std::pair<std::string,std::string>,T>::value ||
231  std::is_same<std::pair<std::string,std::string>,T>::value,
232  "T must be pair<std::string,std::string> or publicly inherit from that type" );
233  unsigned len = 0;
234  if (!IsAlignment(data))
235  return Sequence::SEQMAXUNSIGNED;
236 
237  for (size_t j = 0; j < data[0].second.length (); ++j)
238  {
239  bool site_gapped = 0;
240  for (size_t i = 0; i < data.size (); ++i)
241  {
242  if (data[i][j] == '-')
243  {
244  site_gapped = 1;
245  i = data.size();
246  }
247  }
248  if (!(site_gapped))
249  ++len;
250  }
251  return len;

255  void RemoveGaps (std::vector <T> &data)
256  /*!
257  Modifies the data vector to remove all positions that
258  contain the gap character'-'.
259  \param data vector<T> to modify
260  */
261  {
262  static_assert( std::is_base_of<std::pair<std::string,std::string>,T>::value ||
263  std::is_same<std::pair<std::string,std::string>,T>::value,
264  "T must be pair<std::string,std::string> or publicly inherit from that type" );
265  size_t i, j,datasize=data.size();
266  size_t length = data[0].second.length ();
267  std::vector < std::string > ungapped_sequences(data.size());
268  bool site_is_gapped;
269 
270  for (i = 0; i < length; ++i)
271  { //iterate over sites
272  for ( j = 0, site_is_gapped = 0;
273  j < datasize; ++j)
274  {
275  if (data[j].second[i] == '-')
276  {
277  site_is_gapped = 1;
278  j = datasize;
279  }
280  }
281  if (!(site_is_gapped))
282  {
283  for ( j = 0 ; j != data.size(); ++j)
284  ungapped_sequences[j] += data[j].second[i];
285  }
286  }
287 
288  //redo the data
289  for (j = 0; j != datasize ; ++j)
290  {
291  data[j].second = std::move(ungapped_sequences[j]);
292  }

297  template < typename T >
298  void RemoveTerminalGaps (std::vector <T>&data)
299  /*!
300  Remove all gapped sites from the ends of the alignment,
301  up until the first site on either side that is ungapped.
302  \param data vector<T> to modify
303  */
304  {
305  static_assert( std::is_base_of<std::pair<std::string,std::string>,T>::value ||
306  std::is_same<std::pair<std::string,std::string>,T>::value,
307  "T must be pair<std::string,std::string> or publicly inherit from that type" );
308 
309  size_t i,j,length = data[0].second.length (); //how much we have to iterate over
310  std::vector < std::string > trimmed_sequences;
311  size_t leftmost, rightmost, numUngapped;
312  size_t offset;
313  size_t size = data.size ();
314 
315  leftmost = SEQMAXUNSIGNED;
316  rightmost = length + 1; //offset by one b/c its an array...
317 
318  //find the leftmost site where all sites in the alignment are ungapped
319  for (i = 0; i < length; ++i)
320  { //iterate over sites
321  for (numUngapped = 0, j = 0; j != data.size (); ++j)
322  {
323  if (data[j].second[i] != '-')
324  ++numUngapped;
325  }
326  if (numUngapped == size)
327  {
328  leftmost = i;
329  i = length + 1;
330  }
331  }
332 
333  //find the rigthmost site where all sites in the alignment are ungapped
334  bool exit_condition = false;
335  for (i = length - 1; i < data[0].second.length() && exit_condition == false; --i)
336  {
337  for (numUngapped = 0, j = 0; j != data.size (); ++j)
338  {
339  if (data[j].second[i] != '-')
340  ++numUngapped;
341  }
342  if (numUngapped == size)
343  {
344  rightmost = i;
345  exit_condition = true;
346  }
347  }
348 
349  //now, fill the array of trimmed sequences
350  offset = rightmost - leftmost + 1;
351  for (j = 0; j != data.size (); ++j)
352  trimmed_sequences.push_back (data[j].second.substr (leftmost, offset));
353 
354  //now, redo the seq array for the current object
355  for ( j = 0; j != data.size (); ++j)
356  {
357  //Probably can't hurt...
358  data[j] = std::move(T(data[j].first,trimmed_sequences[j]));
359  }
360  }

399  template < typename T >
400  std::vector < T >Trim (const std::vector < T >&data,
401  const std::vector <int> &sites)
402 
403  /*!
404  Returns a copy of the data vector, modified in the following way. The sites vector
405  contains an even number of sites (whose values are sorted). If sites
406  does not contain an even number of values std::runtime_error is thrown.
407  If sites is empty, std::runtime_error is thrown. The values in sites
408  represent a series of intervals that you wish to keep, and the return vector is
409  consists only of those--i.e. all positions not present
410  in the intervals defined in sites are lost. For example, if you pass a
411  vector<int> containing the values 0,10,21, and 30, then the data vector is modified
412  so that positions 0 through 10 and 21 through 30 are all that remains.
413  One intended use of this function is to pull, for example, the coding region
414  out of an aligned block.
415  \param data the original data
416  \param sites vector<int> containing an even number of integers
417  specifying the intervals of data to keep
418  \exception std::runtime_error
419  */
420  {
421  static_assert( std::is_base_of<std::pair<std::string,std::string>,T>::value ||
422  std::is_same<std::pair<std::string,std::string>,T>::value,
423  "T must be pair<std::string,std::string> or publicly inherit from that type" );
424  size_t i, j, numseqs = data.size (), numIntervals = sites.size ();
425  std::vector < T >trimmedData(numseqs);
426  std::vector < std::string > trimmedTemp(numseqs);
427  if (sites.empty ())
428  {
429  throw std::runtime_error ("Sequence::Alignment::Trim(): empty vector of positions passed to function");
430  }
431  if (numIntervals % 2 != 0)
432  {
433  throw std::runtime_error ("Sequence::Alignment::Trim(): odd number of positions passed");
434  }
435 
436  for (i = 0; i < numIntervals; i += 2)
437  {
438  unsigned start = unsigned(sites[i]);
439  unsigned stop = unsigned(sites[i + 1]);
440  for (j = 0; j < numseqs; ++j)
441  {
442  trimmedTemp[j] += data[j].second.substr (start, stop - start + 1);
443  }
444  }
445 
446  for (i = 0; i < numseqs; ++i)
447  {
448  trimmedData[i]=T(data[i].first,trimmedTemp[i]);
449  }
450 
451  return trimmedData;
452  }

456  const std::vector < int > &sites)

459  /*!
460  Returns a copy the data vector, modified in the following way. The sites vector
461  contains an even number of sites (whose values are sorted). If sites
462  does not contain an even number of values std::runtime_error is thrown.
463  If sites is empty, std::runtime_error is thrown. The values in sites
464  represent a series of intervals that you wish to keep, and the return vector
465  consists only of sites not present in \a sites--i.e. all positions not present
466  in the intervals defined in sites are kept. For example, if you pass a
467  vector<int> containing the values 0,10,21, and 30, then the data vector is modified
468  so that positions 11 through 20 and 31 through the end of the sequences
469  are all that remains.
470  \param data the original data
471  \param sites vector<int> containing an even number of integers
472  specifying the intervals of data to throw away
473  \exception std::runtime_error
474  */
475  {
476  std::vector < int >newSites;
477  size_t i, j, numseqs = data.size (), numIntervals = sites.size (), lastval;
478 
479  if (sites.empty ())
480  {
481  throw std::runtime_error ("Sequence::Alignment::TrimComplement(): empty vector of positions passed to function");
482  }
483  if (numIntervals % 2 != 0)
484  {
485  throw std::runtime_error ("Sequence::Alignment::TrimComplement(): odd numer of positions passed to function");
486  }
487 
488  std::vector < T >trimmedData(numseqs);
489  std::vector < std::string > trimmedTemp(numseqs);
490 
491  size_t odd_even;
492  if (sites[0] == 0)
493  {
494  for (i = 1; i < numIntervals; ++i)
495  {
496  odd_even = i+1;
497  if (odd_even%2==0)
498  {
499  newSites.push_back (sites[i] + 1);
500  }
501  else if (odd_even%2!=0)
502  {
503  newSites.push_back (sites[i] - 1);
504  }
505  }
506  }
507  else if (sites[0] > 0)
508  {
509  newSites.push_back (0);
510  for (i = 0; i < numIntervals; ++i)
511  {
512  odd_even = i+1;
513  if (odd_even%2==0)
514  {
515  newSites.push_back (sites[i] + 1);
516  }
517  else if (odd_even%2!=0)
518  {
519  newSites.push_back (sites[i] - 1);
520  }
521  }
522  }
523 
524  lastval = size_t(newSites[newSites.size () - 1]);
525  newSites.pop_back ();
526  numIntervals = newSites.size ();
527  for (i = 0; i < numIntervals; i += 2)
528  {
529  auto start = size_t(newSites[i]);
530  auto stop = size_t(newSites[i + 1]);
531  for (j = 0; j < numseqs; ++j)
532  {
533  trimmedTemp[j] +=
534  data[j].second.substr (start, stop - start + 1);
535  }
536  }
537 
538  for (j = 0; j < numseqs; ++j)
539  {
540  trimmedTemp[j] += data[j].second.substr (lastval);
541  }
542 
543  for (i = 0; i < numseqs; ++i)
544  {
545  trimmedData[i] = T(data[i].first,trimmedTemp[i]);
