20 double pi( const Sequence::SimData & d )
21 {
22  unsigned ndiffs=0,ncomps=0;
23 
24  for( unsigned i = 0 ; i < d.size() - 1 ; ++i )
25  {
26  for( unsigned j = i + 1 ; j < d.size() ; ++j )
27  {
28  ++ncomps;
29  for( unsigned k = 0 ; k < d[i].size() ; ++k )
30  {
31  ndiffs += (d[i][k] != d[j][k]) ? 1 : 0;
32  }
33  }
34  }
35  return double(ndiffs)/double(ncomps);
36 }

38 unsigned nsingletons( const Sequence::SimData & d )
39 {
40  if(d.empty())return 0;
41  unsigned nsing = 0;
42  std::for_each(d.sbegin(),d.send(),
43  [&nsing,&d](const Sequence::polymorphicSite & __ps) {
44  auto c = std::count(__ps.second.begin(),__ps.second.end(),'1');
45  nsing += ( unsigned(c) == 1 || unsigned(c) == d.size() - 1 ) ? 1 : 0;
46  });

188  auto c = std::count( d.begin(),d.end(), h );
189  hdiv += ( double(c)/double(d.size()) )*(double(d.size()-c)/double(d.size()-1));
190  hhom += ( double(c)/double(d.size()) )*(double(c-1)/double(d.size()-1));
191  }
192  double hdiv2 = 1. - hhom;
193  double lseqvalue = ad.DandVH();
194  BOOST_REQUIRE_CLOSE( lseqvalue, hdiv, 1e-3);
195  BOOST_REQUIRE_CLOSE( lseqvalue, hdiv2, 1e-3);
