260  void bamrecordImpl::parse_block()
261  {
262  __refID= std::move(toItype<int32_t>(__alnblock.get()));
263  __pos = std::move(toItype<int32_t>(__alnblock.get()+sizeof(int32_t)));
264  __bin_mq_nl = std::move(toItype<uint32_t>(__alnblock.get()+2*sizeof(int32_t)));
265  __bin = __bin_mq_nl>>16;
266  __mq = ((__bin_mq_nl ^ __bin<<16)>>8 );
267  __nl= __bin_mq_nl ^ __bin<<16 ^ __mq << 8 ;
268  __flag_nc = std::move(toItype<uint32_t>(__alnblock.get()+2*sizeof(int32_t)+sizeof(uint32_t)));
269  __flag = __flag_nc >> 16;
270  __nc = __flag_nc ^ __flag << 16;
271  __l_seq = std::move(toItype<int32_t>(__alnblock.get()+2*(sizeof(int32_t)+sizeof(uint32_t))));
272  __next_refID = std::move(toItype<int32_t>(__alnblock.get()+3*sizeof(int32_t)+2*sizeof(uint32_t)));
273  __next_pos = std::move(toItype<int32_t>(__alnblock.get()+4*sizeof(int32_t)+2*sizeof(uint32_t)));
274  __tlen = std::move(toItype<int32_t>(__alnblock.get()+5*sizeof(int32_t)+2*sizeof(uint32_t)));
275  __rname_beg = __alnblock.get()+6*sizeof(int32_t)+2*sizeof(uint32_t);
276  __rname_end = __alnblock.get()+6*sizeof(int32_t)+2*sizeof(uint32_t)+__nl;
277  __cig_beg = (uint32_t*)(__alnblock.get()+6*sizeof(int32_t)+2*sizeof(uint32_t)+__nl);
278  __cig_end = (uint32_t*)(__alnblock.get()+6*sizeof(int32_t)+2*sizeof(uint32_t)+__nl + __nc*sizeof(uint32_t));
279  __seq_beg = (uint8_t*)(__alnblock.get()+6*sizeof(int32_t)+2*sizeof(uint32_t)+__nl + __nc*sizeof(uint32_t));
280  __seq_end = (uint8_t*)(__alnblock.get()+6*sizeof(int32_t)+2*sizeof(uint32_t)+__nl + __nc*sizeof(uint32_t) + (__l_seq+1)/2);
